Met de p5.sound library kun je geluiden afspelen, opnemen en analyseren in je sketch. Maak audio-visualisaties, interactieve muziek of gebruik geluid als input voor je animaties.

p5.sound Laden

De p5.sound library is een apart bestand dat je naast p5.js moet laden. Voeg het toe aan je HTML vóór je sketch:

<!-- Voeg toe in je HTML, na p5.js -->
<script src="https://cdnjs.cloudflare.com/ajax/libs/p5.js/1.7.0/addons/p5.sound.min.js"></script>

Let op: in de live editor op deze site is p5.sound niet geladen. De codevoorbeelden hieronder zijn bedoeld voor gebruik in je eigen HTML-projecten.

Een Geluid Afspelen

Laad een geluidsbestand met loadSound() in preload() en speel het af met .play().

let geluid;

function preload() {
  geluid = loadSound('mijn-geluid.mp3');
}

function setup() {
  createCanvas(400, 300);
  background(220);
}

function mousePressed() {
  if (geluid.isPlaying()) {
    geluid.stop();
  } else {
    geluid.play();
  }
}

Geluidsopties

// Volume instellen (0.0 = stil, 1.0 = vol)
geluid.setVolume(0.5);

// Afspelen in een lus
geluid.loop();

// Stoppen en terugspoelen
geluid.stop();

// Pauzeren (hervat later op dezelfde positie)
geluid.pause();

// Hoe lang duurt het geluid?
let duur = geluid.duration(); // in seconden

// Op welke positie zit het afspelen nu?
let positie = geluid.currentTime();

Ondersteunde bestandsformaten

  • .mp3 — werkt in alle moderne browsers
  • .wav — hoge kwaliteit, grote bestanden
  • .ogg — open formaat, goede compressie
  • .m4a — werkt goed in Safari

Tip: bied zowel .mp3 als .ogg aan voor de beste browsercompatibiliteit.

Amplitude: Hoe Hard?

Met de p5.Amplitude analyzer meet je het volume (de amplitude) van een geluid. Gebruik dit voor visualisaties die reageren op hoe hard het geluid is.

let geluid;
let analyzer;

function preload() {
  geluid = loadSound('muziek.mp3');
}

function setup() {
  createCanvas(600, 400);
  geluid.loop();

  analyzer = new p5.Amplitude();
  analyzer.setInput(geluid); // analyseer dit geluid
}

function draw() {
  background(20);

  // Haal het volume op (0.0 - 1.0)
  let volume = analyzer.getLevel();

  // Teken een cirkel die groter wordt naarmate het harder is
  let diameter = map(volume, 0, 1, 10, 400);
  fill(100, 200, 255);
  noStroke();
  circle(width / 2, height / 2, diameter);

  fill(255);
  textSize(16);
  textAlign(CENTER);
  text('Volume: ' + nf(volume, 1, 2), width / 2, 30);
}

Smoothing: vloeiende animatie

// Smoothing maakt de amplitude-waarden vloeiender (0.0 - 1.0)
// 0.0 = direct reageren, 0.9 = langzaam reageren
analyzer.smooth(0.8);

FFT: Frequentie Analyse

Met p5.FFT analyseer je de frequentie-inhoud van geluid: welke tonen zitten er in? Gebruik dit voor equalizer-visualisaties en spectrum displays.

let geluid;
let fft;

function preload() {
  geluid = loadSound('muziek.mp3');
}

function setup() {
  createCanvas(600, 300);
  geluid.loop();

  fft = new p5.FFT();
  fft.setInput(geluid);
}

function draw() {
  background(20);

  // Haal het spectrum op (array van 1024 waarden, van laag naar hoog)
  let spectrum = fft.analyze();

  noStroke();
  fill(100, 200, 255);

  // Teken een staafdiagram van het frequentiespectrum
  for (let i = 0; i < spectrum.length; i++) {
    let x = map(i, 0, spectrum.length, 0, width);
    let hoogte = map(spectrum[i], 0, 255, 0, height);
    rect(x, height - hoogte, width / spectrum.length, hoogte);
  }
}

Specifieke frequentiebanden

// Haal de energie in een frequentieband op
let bass    = fft.getEnergy('bass');     // lage tonen
let mid     = fft.getEnergy('mid');      // middentonen
let treble  = fft.getEnergy('treble');   // hoge tonen

// Of een specifiek bereik (in Hz)
let energie = fft.getEnergy(200, 400);

Microfoon Input

Je kunt ook het geluid van de microfoon van de gebruiker analyseren. De browser vraagt om toestemming.

let mic;
let analyzer;

function setup() {
  createCanvas(600, 400);

  // Microfoon aansluiten
  mic = new p5.AudioIn();
  mic.start();

  // Microfoon analyseren
  analyzer = new p5.Amplitude();
  analyzer.setInput(mic);
}

function draw() {
  background(20);

  let volume = analyzer.getLevel();
  let diameter = map(volume, 0, 1, 10, min(width, height) - 20);

  fill(255, 100, 50);
  noStroke();
  circle(width / 2, height / 2, diameter);

  fill(255);
  textAlign(CENTER);
  textSize(14);
  text('Praat of fluit om de cirkel groter te maken', width / 2, 30);
}

Geluiden Genereren (Oscillator)

Je kunt ook zelf tonen genereren met een p5.Oscillator. Dit werkt zonder geluidsbestand.

let osc;
let aanGezet = false;

function setup() {
  createCanvas(400, 300);
  background(220);

  osc = new p5.Oscillator('sine'); // 'sine', 'triangle', 'square', 'sawtooth'
  osc.amp(0.5);    // volume 50%
  osc.freq(440);   // frequentie in Hz (440 = la)
}

function mousePressed() {
  if (!aanGezet) {
    osc.start();
    aanGezet = true;
  } else {
    osc.stop();
    aanGezet = false;
  }
}

function mouseMoved() {
  // Muis links/rechts bepaalt de frequentie
  let freq = map(mouseX, 0, width, 100, 1000);
  osc.freq(freq);

  // Muis omhoog/omlaag bepaalt het volume
  let vol = map(mouseY, 0, height, 1.0, 0.0);
  osc.amp(vol);
}

Verschillende golfvormen

// Golfvormen hebben elk een eigen klankkleur:
osc = new p5.Oscillator('sine');     // zachte, pure toon
osc = new p5.Oscillator('triangle'); // iets scherpere toon
osc = new p5.Oscillator('square');   // elektronische, buzzende toon
osc = new p5.Oscillator('sawtooth'); // heldere, metaalachtige toon

Audio Visualisatie: Volledig Voorbeeld

Een volledig voorbeeld dat spectrum en amplitude combineert:

let geluid;
let fft;
let amplitude;

function preload() {
  geluid = loadSound('muziek.mp3');
}

function setup() {
  createCanvas(600, 400);
  geluid.loop();

  fft = new p5.FFT(0.8, 64); // smoothing 0.8, 64 banden
  amplitude = new p5.Amplitude(0.8);
}

function draw() {
  background(10, 10, 30, 80); // semi-transparant voor sleep effect

  let spectrum = fft.analyze();
  let vol = amplitude.getLevel();

  // Teken spectrum als cirkels
  noFill();
  for (let i = 0; i < spectrum.length; i++) {
    let hoek = map(i, 0, spectrum.length, 0, TWO_PI);
    let r = map(spectrum[i], 0, 255, 50, 200);
    let x = width / 2 + cos(hoek) * r;
    let y = height / 2 + sin(hoek) * r;

    let kleur = map(i, 0, spectrum.length, 0, 255);
    stroke(kleur, 200 - kleur, 255);
    strokeWeight(2);
    point(x, y);
  }

  // Pulserende cirkel op basis van volume
  noStroke();
  fill(255, 200, 50, 100);
  let diameter = map(vol, 0, 1, 20, 150);
  circle(width / 2, height / 2, diameter);
}

function mousePressed() {
  if (geluid.isPlaying()) {
    geluid.pause();
  } else {
    geluid.play();
  }
}
