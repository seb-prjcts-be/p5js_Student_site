Perlin noise is een speciale wiskundige functie die vloeiende, "organische" willekeur genereert. Waar random() elk frame een compleet willekeurige waarde geeft, geeft noise() waarden die rustig in elkaar overlopen — ideaal voor vloeiende beweging, texturen en landschappen.

noise() vs random()

Het verschil is meteen zichtbaar:

// random(): springt elke frame naar een willekeurige waarde
let x = random(0, width);   // Schokkerig, onrustig

// noise(): vloeiende overgang tussen waarden
let offset = 0;

function draw() {
  let x = noise(offset) * width;  // noise() geeft waarde tussen 0 en 1
  offset += 0.01;                 // Kleine stap = vloeiende beweging
}

noise(t) geeft altijd een waarde tussen 0 en 1. Door t langzaam te verhogen, beweeg je door de noise-ruimte.

Basisgebruik

let t = 0;

function draw() {
  background(245);

  let x = noise(t) * width;
  let y = noise(t + 100) * height;  // Ander startpunt = andere beweging

  circle(x, y, 40);

  t += 0.01;  // Snelheid: kleiner = trager en vloeiender
}

Door t + 100 voor de y-coördinaat te gebruiken, zorgen we dat x en y onafhankelijk van elkaar bewegen.

noise() Mappen naar een Bereik

noise() geeft altijd een waarde tussen 0 en 1. Gebruik map() om dat om te zetten naar een gewenst bereik:

let t = 0;

function draw() {
  let diameter = map(noise(t), 0, 1, 20, 200);
  // of: let diameter = noise(t) * 200;

  background(245);
  circle(width/2, height/2, diameter);
  t += 0.02;
}

2D Noise: Texturen

noise() werkt ook in twee dimensies: noise(x, y). Zo maak je naadloze texturen en "wolken":

function setup() {
  createCanvas(400, 400);
  noLoop();  // Slechts één keer tekenen
}

function draw() {
  loadPixels();
  let schaal = 0.005;  // Kleiner = grotere "vlekken"

  for (let x = 0; x < width; x++) {
    for (let y = 0; y < height; y++) {
      let n = noise(x * schaal, y * schaal);
      let helderheid = n * 255;
      let index = (x + y * width) * 4;
      pixels[index]     = helderheid;  // R
      pixels[index + 1] = helderheid;  // G
      pixels[index + 2] = helderheid;  // B
      pixels[index + 3] = 255;         // Alpha
    }
  }
  updatePixels();
}

noiseSeed() en noiseDetail()

Je kunt het noise-patroon beïnvloeden:

// noiseSeed(): zelfde seed = altijd hetzelfde patroon
noiseSeed(42);

// noiseDetail(): hoeveel "lagen" van detail (octaven)
// Meer octaven = meer detail maar grilliger
noiseDetail(8, 0.5);  // 8 octaven, amplitude halveert per octaaf

Noise voor Vloeiende Beweging

Een klassieker: meerdere objecten die elk onafhankelijk door de noise-ruimte bewegen:

let offsets = [];
let N = 20;

function setup() {
  createCanvas(600, 400);
  // Geef elk object een ander startpunt in de noise-ruimte
  for (let i = 0; i < N; i++) {
    offsets.push({ x: random(1000), y: random(1000) });
  }
}

function draw() {
  background(30, 20);

  noStroke();
  fill(0, 0, 255, 180);

  for (let off of offsets) {
    let x = noise(off.x) * width;
    let y = noise(off.y) * height;

    circle(x, y, 15);

    off.x += 0.007;  // Beweeg door de noise-ruimte
    off.y += 0.007;
  }
}

Noise Landschap

Noise is perfect voor terrein en golven:

let t = 0;

function draw() {
  background(30, 100, 180);  // Lucht

  fill(50, 180, 80);  // Groen terrein
  noStroke();
  beginShape();
  vertex(0, height);

  for (let x = 0; x <= width; x += 5) {
    let y = noise(x * 0.003 + t) * height * 0.7 + height * 0.2;
    vertex(x, y);
  }

  vertex(width, height);
  endShape(CLOSE);

  t += 0.003;  // Laat terrein langzaam schuiven
}
