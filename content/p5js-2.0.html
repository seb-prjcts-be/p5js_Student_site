p5.js 2.0 is een grote update met nieuwe features, verbeterde performance en een paar wijzigingen ten opzichte van versie 1.x. Op deze pagina leer je wat er nieuw is en hoe je de nieuwe mogelijkheden gebruikt.

Migratie van 1.x naar 2.0

De meeste p5.js 1.x code werkt ook in 2.0, maar er zijn een paar belangrijke wijzigingen waar je rekening mee moet houden:

Wat is veranderd?

  • Async asset loading — loadImage() en loadFont() werken nu met async/await in setup()
  • Pointer events — nieuwe, universele events die werken voor zowel muis als touchscreen
  • Verbeterde typografie — meer controle over fonts en tekst rendering
  • Betere curves — verbeterde beginShape() en spline functies
  • WebGL verbeteringen — betere shader ondersteuning en 3D performance

Wat werkt nog hetzelfde?

Alle basisprincipes blijven hetzelfde: setup(), draw(), vormen tekenen, kleuren, animatie — dat alles werkt gewoon door. De meeste 1.x sketches draaien zonder aanpassingen in 2.0.

// Dit werkt in zowel 1.x als 2.0
function setup() {
  createCanvas(600, 400);
  background(220);
}

function draw() {
  fill(mouseX, 100, 200);
  circle(mouseX, mouseY, 50);
}

Async Assets Laden

In p5.js 1.x gebruikte je preload() om afbeeldingen en fonts te laden vóór setup(). In p5.js 2.0 kun je dit ook doen met async/await direct in setup().

De oude manier (1.x — werkt nog steeds)

let img;

function preload() {
  img = loadImage('foto.jpg'); // blokkeert totdat geladen
}

function setup() {
  createCanvas(600, 400);
}

function draw() {
  image(img, 0, 0);
}

De nieuwe manier (2.0 — met async/await)

let img;
let lettertype;

async function setup() {
  createCanvas(600, 400);

  // Laad meerdere assets tegelijk
  img = await loadImage('foto.jpg');
  lettertype = await loadFont('mijn-font.ttf');

  textFont(lettertype);
  background(220);
}

function draw() {
  image(img, 0, 0);
  fill(255);
  textSize(32);
  text('Hallo!', 20, 50);
}

Het voordeel van async/await is dat je meerdere assets kunt laden en foutafhandeling kunt toevoegen:

async function setup() {
  createCanvas(600, 400);

  try {
    img = await loadImage('foto.jpg');
  } catch (fout) {
    console.error('Afbeelding kon niet worden geladen:', fout);
    // Toon een vervanger als de afbeelding niet laadt
    background(200, 100, 100);
    fill(255);
    text('Afbeelding niet gevonden', 20, 200);
  }
}

// Je kunt ook meerdere assets tegelijk laden (sneller!)
async function setup() {
  createCanvas(600, 400);
  [img1, img2, img3] = await Promise.all([
    loadImage('foto1.jpg'),
    loadImage('foto2.jpg'),
    loadImage('foto3.jpg')
  ]);
}

Pointer Events

In p5.js 1.x had je aparte events voor muis (mousePressed()) en touch (touchStarted()). In p5.js 2.0 zijn er nieuwe pointer events die werken voor beide — ideaal voor mobiel én desktop.

Pointer variabelen

function draw() {
  background(220);

  // pointerX en pointerY werken voor muis EN touch
  fill(255, 100, 0);
  circle(pointerX, pointerY, 60);

  // pointerIsDown: true als muis ingedrukt of vinger op scherm
  if (pointerIsDown) {
    fill(0, 200, 100);
    circle(pointerX, pointerY, 100);
  }
}

Pointer event functies

// Wordt aangeroepen bij muisklik OF touch start
function pointerPressed() {
  fill(random(255), random(255), random(255));
  circle(pointerX, pointerY, 40);
}

// Wordt aangeroepen bij bewegen (muis of vinger)
function pointerMoved() {
  stroke(0, 150, 255);
  strokeWeight(3);
  line(ppointerX, ppointerY, pointerX, pointerY); // ppointerX/Y = vorige positie
}

// Wordt aangeroepen bij loslaten
function pointerReleased() {
  console.log('Losgelaten op:', pointerX, pointerY);
}

Vergelijking: oud vs nieuw

// OUDE manier (1.x) — apart voor muis en touch
function mousePressed() { tekenPunt(); }
function touchStarted() { tekenPunt(); return false; }

// NIEUWE manier (2.0) — één functie voor alles
function pointerPressed() { tekenPunt(); }

Let op: de oude mousePressed() en touchStarted() functies werken in 2.0 nog steeds. Je hoeft bestaande code niet aan te passen.

Verbeterde Curves en Splines

p5.js 2.0 heeft verbeterde ondersteuning voor het tekenen van vloeiende curves via beginShape() en endShape().

Catmull-Rom splines met curveVertex()

// Vloeiende curve door controlepunten
beginShape();
curveVertex(50, 300);   // eerste controlepunt (niet getekend)
curveVertex(150, 100);
curveVertex(250, 250);
curveVertex(350, 80);
curveVertex(450, 200);
curveVertex(550, 300);  // laatste controlepunt (niet getekend)
endShape();

Bézier curves in een shape

beginShape();
vertex(50, 200);                         // startpunt
bezierVertex(100, 50, 200, 50, 250, 200); // c1, c2, eindpunt
bezierVertex(300, 350, 400, 350, 450, 200);
endShape();

Kwadratische curves

beginShape();
vertex(50, 300);                    // startpunt
quadraticVertex(200, 50, 350, 300); // controlepunt, eindpunt
quadraticVertex(450, 450, 550, 300);
endShape();

Gesloten shapes met curves

// CLOSE sluit de shape automatisch
beginShape();
curveVertex(300, 100);
curveVertex(400, 200);
curveVertex(350, 350);
curveVertex(200, 350);
curveVertex(150, 200);
endShape(CLOSE);

Nieuwe Typografie Features

p5.js 2.0 heeft verbeterde text rendering, vooral voor webfonts en complexe lettertypes.

Tekst tekenen — basisprincipes

function setup() {
  createCanvas(600, 400);
}

function draw() {
  background(240);

  // Grootte en stijl instellen
  textSize(48);
  textStyle(BOLD);       // NORMAL, BOLD, ITALIC, BOLDITALIC
  textAlign(CENTER, CENTER);

  fill(50);
  text('Hallo p5.js 2.0!', width / 2, height / 2);
}

Tekst opmaken

textSize(24);
textLeading(36);   // regelafstand
textAlign(LEFT);

// Tekst in een rechthoek (automatische regelterugloop)
text('Dit is een lange zin die automatisch afbreekt.', 50, 50, 300, 200);

Een custom font laden

let mijnFont;

async function setup() {
  createCanvas(600, 400);
  mijnFont = await loadFont('mijn-lettertype.ttf');
  textFont(mijnFont);
}

function draw() {
  background(240);
  textSize(64);
  fill(100, 0, 200);
  text('Custom font!', 50, 200);
}

Tekst meten

// Hoe breed is een tekst?
let breedte = textWidth('Hallo');

// Tekst centreren op basis van werkelijke breedte
let woord = 'Gecentreerd';
text(woord, width / 2 - textWidth(woord) / 2, height / 2);

WebGL Verbeteringen (3D)

p5.js 2.0 heeft verbeterde WebGL ondersteuning voor 3D sketches. De basisprincipes zijn hetzelfde, maar de performance is beter en shaders zijn makkelijker in te zetten.

Custom shaders in 2.0

let mijnShader;

async function setup() {
  createCanvas(600, 400, WEBGL);

  // Laad vertex en fragment shaders
  mijnShader = await loadShader('shader.vert', 'shader.frag');
}

function draw() {
  shader(mijnShader);
  mijnShader.setUniform('tijd', millis() / 1000.0);
  rect(0, 0, width, height);
}

Betere 3D performance

function setup() {
  createCanvas(600, 400, WEBGL);
}

function draw() {
  background(20);
  lights();

  // Draaiende bollen — in 2.0 soepeler dankzij verbeterde rendering
  for (let i = 0; i < 5; i++) {
    push();
    translate(i * 100 - 200, 0, 0);
    rotateY(frameCount * 0.02 + i);
    specularMaterial(255, 100, 50);
    sphere(40);
    pop();
  }
}
