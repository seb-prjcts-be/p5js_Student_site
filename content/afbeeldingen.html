Met p5.js kun je afbeeldingen laden en op het canvas tekenen. Je kunt ze schalen, kleuren aanpassen, filters toepassen en zelfs losse pixels bewerken. Dit opent de deur naar creatieve beeldbewerking rechtstreeks in de browser.

Een Afbeelding Laden en Tekenen

Gebruik in moderne p5.js meestal async function setup() met await loadImage(). Zo weet je zeker dat de afbeelding geladen is vóór je ze tekent. In oudere sketches kom je preload() nog tegen.

let foto;

async function setup() {
  foto = await loadImage('mijn-foto.jpg'); // laad de afbeelding
  createCanvas(600, 400);
}

function draw() {
  background(245);
  image(foto, 0, 0); // teken op positie (0, 0)
}

Positie en formaat

Je kunt een afbeelding op een specifieke positie zetten en het formaat aanpassen:

// image(afbeelding, x, y)
image(foto, 50, 50);

// image(afbeelding, x, y, breedte, hoogte)
image(foto, 50, 50, 300, 200);  // schaal naar 300×200

// Afbeelding centreren
imageMode(CENTER);
image(foto, width / 2, height / 2, 200, 150);

Afbeeldingsgrootte opvragen

let foto;

async function setup() {
  foto = await loadImage('mijn-foto.jpg');
  createCanvas(foto.width, foto.height); // canvas even groot als foto
  image(foto, 0, 0);
  console.log('Breedte:', foto.width, 'Hoogte:', foto.height);
}

Tint: Kleur en Transparantie

Met tint() geef je een afbeelding een kleurlaag of maak je hem transparant. Het werkt net als fill(), maar dan voor afbeeldingen.

function draw() {
  background(245);

  // Rood tintje
  tint(255, 0, 0);
  image(foto, 0, 0);

  // Half transparant
  tint(255, 128); // tweede getal = alpha (0-255)
  image(foto, 100, 100);

  // Blauw tintje + half transparant
  tint(0, 100, 255, 180);
  image(foto, 200, 200);

  // Tint uitzetten
  noTint();
  image(foto, 300, 0);
}

let alpha = 0;

function draw() {

Filters

p5.js heeft ingebouwde filters die je op het canvas (of op een afbeelding) kunt toepassen:

function draw() {
  image(foto, 0, 0);

  // Grijswaarden
  filter(GRAY);

  // Omgekeerde kleuren
  filter(INVERT);

  // Vervagen (het getal = sterkte, 1-20)
  filter(BLUR, 5);

  // Drempelwaarde (zwart/wit op basis van helderheid)
  filter(THRESHOLD, 0.5); // waarde 0.0 - 1.0

  // Posteriseren (aantal kleurlagen beperken)
  filter(POSTERIZE, 4); // 2-255
}

Let op: filter() past het filter toe op alles wat al op het canvas staat. Zet het dus altijd ná image().

Pixelmanipulatie

Met loadPixels() en updatePixels() kun je elke individuele pixel van het canvas lezen en aanpassen. Dit is krachtig maar ook langzamer.

Pixels lezen

function setup() {
  createCanvas(400, 400);
  background(100, 150, 200);
  loadPixels(); // laad de pixeldata in het 'pixels' array

  // pixels is een plat array: [R, G, B, A, R, G, B, A, ...]
  // Pixel op positie (x, y):
  let x = 100;
  let y = 50;
  let index = (y * width + x) * 4; // elk pixel = 4 waarden

  let rood  = pixels[index];
  let groen = pixels[index + 1];
  let blauw = pixels[index + 2];
  let alpha = pixels[index + 3];

  console.log('Pixel op (100, 50):', rood, groen, blauw, alpha);
}

Pixels aanpassen

function setup() {
  createCanvas(400, 400);
  image(foto, 0, 0, width, height);

  loadPixels();

  for (let x = 0; x < width; x++) {
    for (let y = 0; y < height; y++) {
      let index = (y * width + x) * 4;

      // Haal de rode waarde op
      let r = pixels[index];
      let g = pixels[index + 1];
      let b = pixels[index + 2];

      // Bereken helderheid (gemiddelde van RGB)
      let helderheid = (r + g + b) / 3;

      // Zet om naar grijswaarden
      pixels[index]     = helderheid; // rood
      pixels[index + 1] = helderheid; // groen
      pixels[index + 2] = helderheid; // blauw
      // pixels[index + 3] = alpha, laten we met rust
    }
  }

  updatePixels(); // sla de wijzigingen op
}

Creatief gebruik van pixels

// Mozaïek effect: vergroot pixels tot blokjes
function setup() {
  createCanvas(400, 400);
  image(foto, 0, 0, width, height);
  loadPixels();

  let blokGrootte = 10;

  for (let x = 0; x < width; x += blokGrootte) {
    for (let y = 0; y < height; y += blokGrootte) {
      let index = (y * width + x) * 4;
      let r = pixels[index];
      let g = pixels[index + 1];
      let b = pixels[index + 2];

      // Teken een gekleurd blokje
      fill(r, g, b);
      noStroke();
      rect(x, y, blokGrootte, blokGrootte);
    }
  }
}

Een Afbeelding Opslaan

Je kunt het canvas als afbeelding opslaan met saveCanvas():

function keyPressed() {
  if (key === 's') {
    saveCanvas('mijn-sketch', 'png'); // opslaan als PNG
    // saveCanvas('mijn-sketch', 'jpg'); // of als JPG
  }
}
