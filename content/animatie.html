Animatie ontstaat doordat draw() steeds opnieuw wordt uitgevoerd - standaard 60 keer per seconde. Door waarden elke frame een beetje te veranderen, krijg je beweging.

Eenvoudige Beweging

De simpelste animatie: een variabele ophogen elke frame.

let x = 0;

function draw() {
  background(245);
  x += 2;         // Beweeg 2 pixels naar rechts per frame
  if (x > width) {
    x = 0;        // Begin opnieuw aan de linkerkant
  }
  circle(x, height / 2, 50);
}

frameCount en millis()

frameCount telt het aantal frames. millis() geeft de verstreken tijd in milliseconden.

function draw() {
  // frameCount loopt oneindig op
  let x = (frameCount * 2) % width;
  circle(x, height / 2, 50);
}

// millis() is handig als framerate kan variëren
function draw() {
  let seconden = millis() / 1000;
  let x = (seconden * 120) % width;  // 120 pixels per seconde
  circle(x, height / 2, 50);
}

Sinus en Cosinus

sin() en cos() geven een waarde tussen -1 en 1 terug die vloeiend op en neer gaat. Ideaal voor organische beweging. Met map() zet je die waarden om naar een handig bereik:

let hoek = 0;

function draw() {
  background(245);

  // sin/cos mappen op een bereik met map()
  let x = width / 2 + cos(hoek) * 150;
  let y = height / 2 + sin(hoek) * 100;

  circle(x, y, 50);
  hoek += 0.03;  // Langzamer = kleinere stap
}

Tangens

tan() is de derde trigonometrische functie. In tegenstelling tot sin/cos loopt tan() naar oneindig - het schiet door het canvas heen en komt aan de andere kant terug. Dat levert grillige, energieke patronen op:

let hoek = 0;

function draw() {
  background(245);
  let x = width / 2 + tan(hoek) * 50;
  let y = height / 2;
  circle(x, y, 20);
  hoek += 0.03;
}

Lissajous-figuur

Door twee verschillende frequenties te combineren, ontstaan mooie Lissajous-figuren:

let hoek = 0;

function draw() {
  // Geen background() → spoor blijft zichtbaar
  let x = width / 2 + cos(hoek * 3) * 150;
  let y = height / 2 + sin(hoek * 2) * 150;
  circle(x, y, 8);
  hoek += 0.02;
}

Easing: Vloeiende Beweging

Easing laat een object vloeiend naar een doel bewegen, in plaats van abrupt teleporteren. Het idee: beweeg elke frame een fractie van de resterende afstand.

let huidigeX = 0;
let doelX = 0;

function draw() {
  background(245);

  // Beweeg 10% dichter naar het doel toe
  huidigeX += (doelX - huidigeX) * 0.1;

  circle(huidigeX, height / 2, 50);
}

function mousePressed() {
  doelX = mouseX;  // Nieuw doel instellen
}

translate() en rotate()

translate(x, y) verschuift het assenstelsel - alsof je het papier onder de pen verplaatst. Daarna teken je gewoon op (0, 0):

function draw() {
  background(245);
  translate(mouseX, mouseY);  // Assenstelsel naar muispositie
  rect(-25, -25, 50, 50);     // Tekent gecentreerd op muispositie
}

rotate(hoek) draait het assenstelsel. p5.js werkt standaard in radialen. Wil je liever graden, zet dan angleMode(DEGREES) in setup():

function setup() {
  createCanvas(600, 400);
  angleMode(DEGREES);  // Werk voortaan in graden
}

function draw() {
  background(245);
  translate(width / 2, height / 2);  // Draai om het midden
  rotate(frameCount);                  // 1 graad per frame
  rect(-40, -40, 80, 80);
}

Gebruik translate() altijd vóór rotate() om een object om zijn eigen middelpunt te draaien, niet om de linkerbovenhoek.

push() en pop() voor Transformaties

push() en pop() isoleren transformaties zodat ze andere objecten niet beïnvloeden:

function draw() {
  background(245);

  // Roterende kubus links
  push();
    translate(150, 200);
    rotate(frameCount * 0.02);
    rect(-40, -40, 80, 80);
  pop();  // Rotatie wordt hier gereset

  // Dit blok wordt NIET meegedraaid
  push();
    translate(400, 200);
    rotate(frameCount * -0.03);
    rect(-40, -40, 80, 80);
  pop();
}

frameRate()

Je kunt de framerate aanpassen:

function setup() {
  createCanvas(600, 400);
  frameRate(30);  // 30 frames per seconde (standaard: 60)
}

function draw() {
  // frameRate() geeft de actuele framerate terug:
  text("FPS: " + round(frameRate()), 10, 20);
}
