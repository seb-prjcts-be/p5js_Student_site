Functions are reusable pieces of code. They make your code more organized, shorter, and easier to maintain. In p5.js you use them all the time: setup() and draw() are functions themselves!

Defining and Calling Functions

You define a function with the keyword function:

// Definition
function drawCircle() {
  ellipse(200, 200, 50, 50);
}

// Calling — do this in draw() or setup()
function draw() {
  background(245);
  drawCircle();  // Execute the function
  drawCircle();  // Again — no need to repeat the code
}

Parameters: Passing Values

Parameters make functions flexible. You define them between the parentheses:

function drawCircle(x, y, diameter) {
  ellipse(x, y, diameter, diameter);
}

function draw() {
  background(245);
  drawCircle(100, 150, 50);   // x=100, y=150, diameter=50
  drawCircle(300, 200, 80);   // Same function, different values
  drawCircle(450, 100, 30);
}

Multiple Parameters with Color

function drawBall(x, y, diameter, r, g, b) {
  fill(r, g, b);
  noStroke();
  ellipse(x, y, diameter, diameter);
}

function draw() {
  background(30);
  drawBall(150, 200, 60, 255, 0,   0);    // red
  drawBall(300, 200, 80, 0,   0,   255);  // blue
  drawBall(450, 200, 40, 0,   255, 0);    // green
}

Return Values

A function can return a calculated value with return:

function calculateDistance(x1, y1, x2, y2) {
  let dx = x2 - x1;
  let dy = y2 - y1;
  return sqrt(dx * dx + dy * dy);  // Pythagoras
}

function draw() {
  let distance = calculateDistance(mouseX, mouseY, width/2, height/2);
  let diameter = map(distance, 0, 300, 10, 150);

  ellipse(width/2, height/2, diameter, diameter);
}

Another example: random color

function randomColor() {
  return color(random(255), random(255), random(255));
}

function mousePressed() {
  fill(randomColor());
  ellipse(mouseX, mouseY, 40, 40);
}

Complex Example: Drawing a Flower

By combining functions you can easily reuse complex shapes:

function drawPetal(x, y, angle, size) {
  push();
    translate(x, y);
    rotate(angle);
    noStroke();
    fill(255, 0, 125, 180);
    ellipse(size * 0.6, 0, size * 1.2, size * 0.6);
  pop();
}

function drawFlower(x, y, size) {
  let petalCount = 8;
  for (let i = 0; i < petalCount; i++) {
    let angle = (TWO_PI / petalCount) * i;
    drawPetal(x, y, angle, size);
  }
  // Center
  fill(255, 255, 0);
  noStroke();
  ellipse(x, y, size * 0.5, size * 0.5);
}

function draw() {
  background(200, 230, 200);
  drawFlower(150, 200, 60);
  drawFlower(350, 150, 80);
  drawFlower(500, 250, 50);
}
