With p5.js you can load images and draw them on the canvas. You can scale them, adjust colors, apply filters, and even edit individual pixels. This opens the door to creative image manipulation right in the browser.

Loading and Drawing an Image

In modern p5.js, typically use async function setup() with await loadImage(). This ensures the image is loaded before you draw it. In older sketches you may still encounter preload().

let photo;

async function setup() {
  photo = await loadImage('my-photo.jpg'); // load the image
  createCanvas(600, 400);
}

function draw() {
  background(245);
  image(photo, 0, 0); // draw at position (0, 0)
}

Position and size

You can place an image at a specific position and adjust its size:

// image(img, x, y)
image(photo, 50, 50);

// image(img, x, y, width, height)
image(photo, 50, 50, 300, 200);  // scale to 300x200

// Center the image
imageMode(CENTER);
image(photo, width / 2, height / 2, 200, 150);

Getting image dimensions

let photo;

async function setup() {
  photo = await loadImage('my-photo.jpg');
  createCanvas(photo.width, photo.height); // canvas same size as photo
  image(photo, 0, 0);
  console.log('Width:', photo.width, 'Height:', photo.height);
}

Tint: Color and Transparency

With tint() you give an image a color overlay or make it transparent. It works just like fill(), but for images.

function draw() {
  background(245);

  // Red tint
  tint(255, 0, 0);
  image(photo, 0, 0);

  // Half transparent
  tint(255, 128); // second value = alpha (0-255)
  image(photo, 100, 100);

  // Blue tint + half transparent
  tint(0, 100, 255, 180);
  image(photo, 200, 200);

  // Turn off tint
  noTint();
  image(photo, 300, 0);
}

Fading an image

let alpha = 0;

function draw() {
  background(0);
  alpha = (sin(frameCount * 0.05) + 1) * 127; // value between 0 and 255
  tint(255, alpha);
  image(photo, 0, 0, width, height);
}

Filters

p5.js has built-in filters that you can apply to the canvas (or to an image):

function draw() {
  image(photo, 0, 0);

  // Grayscale
  filter(GRAY);

  // Inverted colors
  filter(INVERT);

  // Blur (the number = strength, 1-20)
  filter(BLUR, 5);

  // Threshold (black/white based on brightness)
  filter(THRESHOLD, 0.5); // value 0.0 - 1.0

  // Posterize (limit number of color levels)
  filter(POSTERIZE, 4); // 2-255
}

Note: filter() applies the filter to everything already on the canvas. So always place it after image().

Pixel Manipulation

With loadPixels() and updatePixels() you can read and modify every individual pixel of the canvas. This is powerful but also slower.

Reading pixels

function setup() {
  createCanvas(400, 400);
  background(100, 150, 200);
  loadPixels(); // load pixel data into the 'pixels' array

  // pixels is a flat array: [R, G, B, A, R, G, B, A, ...]
  // Pixel at position (x, y):
  let x = 100;
  let y = 50;
  let index = (y * width + x) * 4; // each pixel = 4 values

  let red   = pixels[index];
  let green = pixels[index + 1];
  let blue  = pixels[index + 2];
  let alpha = pixels[index + 3];

  console.log('Pixel at (100, 50):', red, green, blue, alpha);
}

Modifying pixels

function setup() {
  createCanvas(400, 400);
  image(photo, 0, 0, width, height);

  loadPixels();

  for (let x = 0; x < width; x++) {
    for (let y = 0; y < height; y++) {
      let index = (y * width + x) * 4;

      // Get the color values
      let r = pixels[index];
      let g = pixels[index + 1];
      let b = pixels[index + 2];

      // Calculate brightness (average of RGB)
      let brightness = (r + g + b) / 3;

      // Convert to grayscale
      pixels[index]     = brightness; // red
      pixels[index + 1] = brightness; // green
      pixels[index + 2] = brightness; // blue
      // pixels[index + 3] = alpha, leave it alone
    }
  }

  updatePixels(); // save the changes
}

Creative use of pixels

// Mosaic effect: enlarge pixels into blocks
function setup() {
  createCanvas(400, 400);
  image(photo, 0, 0, width, height);
  loadPixels();

  let blockSize = 10;

  for (let x = 0; x < width; x += blockSize) {
    for (let y = 0; y < height; y += blockSize) {
      let index = (y * width + x) * 4;
      let r = pixels[index];
      let g = pixels[index + 1];
      let b = pixels[index + 2];

      // Draw a colored block
      fill(r, g, b);
      noStroke();
      rect(x, y, blockSize, blockSize);
    }
  }
}

Saving an Image

You can save the canvas as an image with saveCanvas():

function keyPressed() {
  if (key === 's') {
    saveCanvas('my-sketch', 'png'); // save as PNG
    // saveCanvas('my-sketch', 'jpg'); // or as JPG
  }
}
