Animation happens because draw() is executed over and over - 60 times per second by default. By changing values slightly each frame, you get motion.

Simple Movement

The simplest animation: increment a variable each frame.

let x = 0;

function draw() {
  background(245);
  x += 2;         // Move 2 pixels to the right per frame
  if (x > width) {
    x = 0;        // Start over at the left side
  }
  circle(x, height / 2, 50);
}

frameCount and millis()

frameCount counts the number of frames. millis() gives the elapsed time in milliseconds.

function draw() {
  // frameCount increases infinitely
  let x = (frameCount * 2) % width;
  circle(x, height / 2, 50);
}

// millis() is useful when framerate may vary
function draw() {
  let seconds = millis() / 1000;
  let x = (seconds * 120) % width;  // 120 pixels per second
  circle(x, height / 2, 50);
}

Sine and Cosine

sin() and cos() return a value between -1 and 1 that smoothly oscillates up and down. Ideal for organic movement. With map() you convert those values to a useful range:

let angle = 0;

function draw() {
  background(245);

  // Map sin/cos to a range with map()
  let x = width / 2 + cos(angle) * 150;
  let y = height / 2 + sin(angle) * 100;

  circle(x, y, 50);
  angle += 0.03;  // Slower = smaller step
}

Tangent

tan() is the third trigonometric function. Unlike sin/cos, tan() goes to infinity - it shoots across the canvas and comes back on the other side. This produces erratic, energetic patterns:

let angle = 0;

function draw() {
  background(245);
  let x = width / 2 + tan(angle) * 50;
  let y = height / 2;
  circle(x, y, 20);
  angle += 0.03;
}

Lissajous figure

By combining two different frequencies, you create beautiful Lissajous figures:

let angle = 0;

function draw() {
  // No background() → trail remains visible
  let x = width / 2 + cos(angle * 3) * 150;
  let y = height / 2 + sin(angle * 2) * 150;
  circle(x, y, 8);
  angle += 0.02;
}

Easing: Smooth Movement

Easing makes an object move smoothly toward a target, instead of teleporting abruptly. The idea: move a fraction of the remaining distance each frame.

let currentX = 0;
let targetX = 0;

function draw() {
  background(245);

  // Move 10% closer to the target
  currentX += (targetX - currentX) * 0.1;

  circle(currentX, height / 2, 50);
}

function mousePressed() {
  targetX = mouseX;  // Set new target
}

translate() and rotate()

translate(x, y) shifts the coordinate system - as if you're moving the paper under the pen. Then you simply draw at (0, 0):

function draw() {
  background(245);
  translate(mouseX, mouseY);  // Move coordinate system to mouse position
  rect(-25, -25, 50, 50);     // Draws centered at mouse position
}

rotate(angle) rotates the coordinate system. p5.js works in radians by default. If you prefer degrees, set angleMode(DEGREES) in setup():

function setup() {
  createCanvas(600, 400);
  angleMode(DEGREES);  // Work in degrees from now on
}

function draw() {
  background(245);
  translate(width / 2, height / 2);  // Rotate around the center
  rotate(frameCount);                  // 1 degree per frame
  rect(-40, -40, 80, 80);
}

Always use translate() before rotate() to rotate an object around its own center, not around the top-left corner.

Watch out: translate() stacks up

Transformations are cumulative. Calling translate(100, 0) twice is the same as calling translate(200, 0) once. This leads to surprises if you don't expect it:

// WRONG: the circle shifts 100px further every frame!
function draw() {
  background(245);
  translate(100, 200);
  circle(0, 0, 40);
  translate(100, 0);   // Stacks on the previous: now at (200, 200)
  circle(0, 0, 40);    // This circle is at x=200, not x=100
}
// RIGHT: with push/pop each transformation starts clean
function draw() {
  background(245);

  push();
    translate(100, 200);
    circle(0, 0, 40);   // At (100, 200)
  pop();

  push();
    translate(200, 200);
    circle(0, 0, 40);   // At (200, 200) - independent
  pop();
}

Remember: without push()/pop(), every translate() and rotate() is an extra shift on top of everything that came before.

push() and pop() for Transformations

push() and pop() isolate transformations so they don't affect other objects:

function draw() {
  background(245);

  // Rotating square on the left
  push();
    translate(150, 200);
    rotate(frameCount * 0.02);
    rect(-40, -40, 80, 80);
  pop();  // Rotation is reset here

  // This block is NOT rotated along
  push();
    translate(400, 200);
    rotate(frameCount * -0.03);
    rect(-40, -40, 80, 80);
  pop();
}

frameRate()

You can adjust the frame rate:

function setup() {
  createCanvas(600, 400);
  frameRate(30);  // 30 frames per second (default: 60)
}

function draw() {
  // frameRate() returns the current frame rate:
  text("FPS: " + round(frameRate()), 10, 20);
}

Exporting your animation

Want to save a still frame or a series of frames? Use saveCanvas() to export a PNG, or combine it with frameCount for frame-by-frame recording:

// Save one frame (e.g. on mouse click)
function mousePressed() {
  saveCanvas('my-frame', 'png');
}

// Save multiple frames (for a GIF or video)
function draw() {
  // ... your animation ...
  saveCanvas('frame-' + frameCount, 'png');
}
