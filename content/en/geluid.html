With the p5.sound library you can play, record, and analyze sounds in your sketch. Create audio visualizations, interactive music, or use sound as input for your animations.

What you will learn
  • Loading p5.sound as an additional script in your HTML
  • Playing, looping and stopping sounds with loadSound()
  • Measuring volume with p5.Amplitude and frequencies with p5.FFT
  • Microphone input through p5.AudioIn for live audio
  • Generating tones with p5.Oscillator in different waveforms

Loading p5.sound

The p5.sound library is a separate file that you need to load alongside p5.js. Add it to your HTML before your sketch:

<!-- Add to your HTML, after p5.js -->
<script src="https://cdn.jsdelivr.net/npm/p5.sound@0.2.0/dist/p5.sound.min.js"></script>
Watch out

p5.sound is not loaded in the live editor on this site. The code examples below are intended for use in your own HTML projects.

Playing a Sound

In modern p5.js you usually load a sound file with await loadSound() inside async function setup(). In older code you may still see this done in preload().

let sound;

async function setup() {
  sound = await loadSound('my-sound.mp3');
  createCanvas(400, 300);
  background(245);
}

function mousePressed() {
  if (sound.isPlaying()) {
    sound.stop();
  } else {
    sound.play();
  }
}

Sound options

// Set volume (0.0 = silent, 1.0 = full)
sound.setVolume(0.5);

// Play in a loop
sound.loop();

// Stop and rewind
sound.stop();

// Pause (resumes later at the same position)
sound.pause();

// How long is the sound?
let duration = sound.duration(); // in seconds

// What position is the playback at now?
let position = sound.currentTime();

Supported file formats

  • .mp3 - works in all modern browsers
  • .wav - high quality, large files
  • .ogg - open format, good compression
  • .m4a - works well in Safari

Tip: provide both .mp3 and .ogg for the best browser compatibility.

Amplitude: How Loud?

With the p5.Amplitude analyzer you measure the volume (amplitude) of a sound. Use this for visualizations that respond to how loud the sound is.

let sound;
let analyzer;

async function setup() {
  sound = await loadSound('music.mp3');
  createCanvas(600, 400);
  sound.loop();

  analyzer = new p5.Amplitude();
  analyzer.setInput(sound); // analyze this sound
}

function draw() {
  background(20);

  // Get the volume (0.0 - 1.0)
  let volume = analyzer.getLevel();

  // Draw a circle that grows as the sound gets louder
  let diameter = map(volume, 0, 1, 10, 400);
  fill(0, 0, 255);
  noStroke();
  circle(width / 2, height / 2, diameter);

  fill(255);
  textSize(16);
  textAlign(CENTER);
  text('Volume: ' + nf(volume, 1, 2), width / 2, 30);
}

Smoothing: smooth animation

// Smoothing makes the amplitude values smoother (0.0 - 1.0)
// 0.0 = respond instantly, 0.9 = respond slowly
analyzer.smooth(0.8);

FFT: Frequency Analysis

With p5.FFT you analyze the frequency content of sound: which tones are present? Use this for equalizer visualizations and spectrum displays.

let sound;
let fft;

async function setup() {
  sound = await loadSound('music.mp3');
  createCanvas(600, 300);
  sound.loop();

  fft = new p5.FFT();
  fft.setInput(sound);
}

function draw() {
  background(20);

  // Get the spectrum (array of 1024 values, from low to high)
  let spectrum = fft.analyze();

  noStroke();
  fill(0, 0, 255);

  // Draw a bar chart of the frequency spectrum
  for (let i = 0; i < spectrum.length; i++) {
    let x = map(i, 0, spectrum.length, 0, width);
    let h = map(spectrum[i], 0, 255, 0, height);
    rect(x, height - h, width / spectrum.length, h);
  }
}

Specific frequency bands

// Get the energy in a frequency band
let bass    = fft.getEnergy('bass');     // low tones
let mid     = fft.getEnergy('mid');      // mid tones
let treble  = fft.getEnergy('treble');   // high tones

// Or a specific range (in Hz)
let energy = fft.getEnergy(200, 400);

Microphone Input

You can also analyze the sound from the user's microphone. The browser will ask for permission.

let mic;
let analyzer;

function setup() {
  createCanvas(600, 400);

  // Connect microphone
  mic = new p5.AudioIn();
  mic.start();

  // Analyze microphone
  analyzer = new p5.Amplitude();
  analyzer.setInput(mic);
}

function draw() {
  background(20);

  let volume = analyzer.getLevel();
  let diameter = map(volume, 0, 1, 10, min(width, height) - 20);

  fill(255, 0, 0);
  noStroke();
  circle(width / 2, height / 2, diameter);

  fill(255);
  textAlign(CENTER);
  textSize(14);
  text('Talk or whistle to make the circle bigger', width / 2, 30);
}

Generating Sounds (Oscillator)

You can also generate tones yourself with a p5.Oscillator. This works without a sound file.

let osc;
let isOn = false;

function setup() {
  createCanvas(400, 300);
  background(245);

  osc = new p5.Oscillator('sine'); // 'sine', 'triangle', 'square', 'sawtooth'
  osc.amp(0.5);    // volume 50%
  osc.freq(440);   // frequency in Hz (440 = A)
}

function mousePressed() {
  if (!isOn) {
    osc.start();
    isOn = true;
  } else {
    osc.stop();
    isOn = false;
  }
}

function mouseMoved() {
  // Mouse left/right controls the frequency
  let freq = map(mouseX, 0, width, 100, 1000);
  osc.freq(freq);

  // Mouse up/down controls the volume
  let vol = map(mouseY, 0, height, 1.0, 0.0);
  osc.amp(vol);
}

Different waveforms

// Each waveform has its own timbre:
osc = new p5.Oscillator('sine');     // soft, pure tone
osc = new p5.Oscillator('triangle'); // slightly sharper tone
osc = new p5.Oscillator('square');   // electronic, buzzy tone
osc = new p5.Oscillator('sawtooth'); // bright, metallic tone

Audio Visualization: Full Example

A complete example that combines spectrum and amplitude:

let sound;
let fft;
let amplitude;

async function setup() {
  sound = await loadSound('music.mp3');
  createCanvas(600, 400);
  sound.loop();

  fft = new p5.FFT(0.8, 64); // smoothing 0.8, 64 bands
  amplitude = new p5.Amplitude(0.8);
}

function draw() {
  background(10, 10, 30, 80); // semi-transparent for trail effect

  let spectrum = fft.analyze();
  let vol = amplitude.getLevel();

  // Draw spectrum as circles
  noFill();
  for (let i = 0; i < spectrum.length; i++) {
    let angle = map(i, 0, spectrum.length, 0, TWO_PI);
    let r = map(spectrum[i], 0, 255, 50, 200);
    let x = width / 2 + cos(angle) * r;
    let y = height / 2 + sin(angle) * r;

    let c = map(i, 0, spectrum.length, 0, 255);
    stroke(c, 200 - c, 255);
    strokeWeight(2);
    point(x, y);
  }

  // Pulsing circle based on volume
  noStroke();
  fill(255, 200, 50, 100);
  let diameter = map(vol, 0, 1, 20, 150);
  circle(width / 2, height / 2, diameter);
}

function mousePressed() {
  if (sound.isPlaying()) {
    sound.pause();
  } else {
    sound.play();
  }
}
