Classes are blueprints for objects. They bundle properties (data) and behavior (functions) into one unit. In p5.js you often use classes for things like balls, particles, or enemies in a game — objects that each keep track of their own state.

From Object Literal to Class

You already knew object literals from arrays:

let ball = { x: 100, y: 200, vx: 2, vy: 0 };

But what if you want dozens of balls, each with their own methods? Then a class is more convenient:

class Ball {
  constructor(x, y) {
    this.x = x;
    this.y = y;
    this.vx = random(-2, 2);
    this.vy = random(-3, -1);
  }

  update() {
    this.x += this.vx;
    this.y += this.vy;
    this.vy += 0.1;  // Gravity
  }

  display() {
    circle(this.x, this.y, 20);
  }
}

Creating Objects with new

You create an instance of a class with new:

let myBall = new Ball(200, 300);  // Calls the constructor

myBall.update();  // Execute a method
myBall.display();

Multiple Objects in an Array

let balls = [];

function setup() {
  createCanvas(600, 400);

  // Create 10 balls
  for (let i = 0; i < 10; i++) {
    balls.push(new Ball(random(width), random(height)));
  }
}

function draw() {
  background(245);

  for (let ball of balls) {
    ball.update();
    ball.display();
  }
}

A Complete Class Example

A ball that bounces against the edges of the canvas:

class BouncingBall {
  constructor(x, y, diameter) {
    this.x = x;
    this.y = y;
    this.d = diameter;
    this.vx = random(-3, 3);
    this.vy = random(-3, 3);
    this.clr = color(random(255), random(255), random(255));
  }

  update() {
    this.x += this.vx;
    this.y += this.vy;

    // Bounce off left/right edge
    if (this.x < this.d / 2 || this.x > width - this.d / 2) {
      this.vx *= -1;
    }
    // Bounce off top/bottom edge
    if (this.y < this.d / 2 || this.y > height - this.d / 2) {
      this.vy *= -1;
    }
  }

  display() {
    fill(this.clr);
    noStroke();
    circle(this.x, this.y, this.d);
  }
}

let balls = [];

function setup() {
  createCanvas(600, 400);
  for (let i = 0; i < 15; i++) {
    balls.push(new BouncingBall(
      random(width), random(height), random(20, 60)
    ));
  }
}

function draw() {
  background(30);
  for (let ball of balls) {
    ball.update();
    ball.display();
  }
}

Methods that Return Values

Methods can also calculate and return values:

class Ball {
  constructor(x, y) {
    this.x = x;
    this.y = y;
    this.d = 30;
  }

  // Returns true if two balls overlap
  overlaps(otherBall) {
    let distance = dist(this.x, this.y, otherBall.x, otherBall.y);
    return distance < (this.d + otherBall.d) / 2;
  }

  display(hit) {
    fill(hit ? 255 : 0, 0, 255);
    circle(this.x, this.y, this.d);
  }
}

let a, b;

function setup() {
  createCanvas(400, 300);
  a = new Ball(100, 150);
  b = new Ball(300, 150);
}

function draw() {
  background(245);
  b.x = mouseX;
  b.y = mouseY;

  let isOverlapping = a.overlaps(b);
  a.display(isOverlapping);
  b.display(isOverlapping);
}

The this Keyword

this always refers to the current object. This is how you distinguish object properties from local variables:

class Counter {
  constructor(startValue) {
    this.value = startValue;  // Property of the object
  }

  increment(step) {
    let step = step || 1;    // Local variable
    this.value += step;      // Works on the object
  }
}
