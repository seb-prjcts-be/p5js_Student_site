Noise is a special mathematical function that generates smooth, "organic" randomness. Where random() gives a completely random value each frame, noise() gives values that transition smoothly into each other - ideal for fluid movement, textures, and landscapes.

Did you know

The noise() you use here is formally called Perlin noise, invented by Ken Perlin in 1983 while he was working on the visual effects of the film Tron (1982). He was looking for a way to create lively textures that didn't look digital or blocky. His invention became a standard technique in games, films, and generative art. In 1997 he received an Academy Award for Technical Achievement for it.

What you will learn
  • The difference between random() (jerky) and noise() (smooth)
  • noise(t) always returns a value between 0 and 1 - step slowly through t for smooth motion
  • Mapping values to a useful range with map()
  • 2D noise noise(x, y) for textures, clouds and landscapes
  • noiseSeed() for repeatable patterns and noiseDetail() for extra octaves

noise() vs random()

The difference is immediately visible:

// random(): jumps to a random value every frame
let x = random(0, width);   // Jerky, restless

// noise(): smooth transition between values
let offset = 0;

function draw() {
  let x = noise(offset) * width;  // noise() returns value between 0 and 1
  offset += 0.01;                 // Small step = smooth motion
}

noise(t) always returns a value between 0 and 1. By slowly increasing t, you move through the noise space.

Basic usage

let t = 0;

function draw() {
  background(245);

  let x = noise(t) * width;
  let y = noise(t + 100) * height;  // Different starting point = different motion

  circle(x, y, 40);

  t += 0.01;  // Speed: smaller = slower and smoother
}

By using t + 100 for the y-coordinate, we ensure that x and y move independently of each other.

Mapping noise() to a Range

noise() always returns a value between 0 and 1. Use map() to convert it to a desired range:

let t = 0;

function draw() {
  let diameter = map(noise(t), 0, 1, 20, 200);
  // or: let diameter = noise(t) * 200;

  background(245);
  circle(width/2, height/2, diameter);
  t += 0.02;
}

2D Noise: Textures

noise() also works in two dimensions: noise(x, y). This way you create seamless textures and "clouds":

function setup() {
  createCanvas(400, 400);
  noLoop();  // Draw only once
}

function draw() {
  loadPixels();
  let scale = 0.005;  // Smaller = larger "blobs"

  for (let x = 0; x < width; x++) {
    for (let y = 0; y < height; y++) {
      let n = noise(x * scale, y * scale);
      let brightness = n * 255;
      let index = (x + y * width) * 4;
      pixels[index]     = brightness;  // R
      pixels[index + 1] = brightness;  // G
      pixels[index + 2] = brightness;  // B
      pixels[index + 3] = 255;         // Alpha
    }
  }
  updatePixels();
}

noiseSeed() and noiseDetail()

You can influence the noise pattern:

// noiseSeed(): same seed = always the same pattern
noiseSeed(42);

// noiseDetail(): how many "layers" of detail (octaves)
// More octaves = more detail but more erratic
noiseDetail(8, 0.5);  // 8 octaves, amplitude halves per octave

Flow field: noise as directions

The best-known use of noise in generative art is a flow field. The idea is simple: at every point (x, y) on the canvas we ask for a noise value and use it as an angle. Particles follow those angles, as if drifting on an invisible current.

let particles = [];
let scale = 0.005;  // Smaller = bigger swirls

function setup() {
  createCanvas(600, 400);
  background(10);
  for (let i = 0; i < 200; i++) {
    particles.push({ x: random(width), y: random(height) });
  }
}

function draw() {
  background(10, 15);  // Trails fade

  stroke(255, 150);
  strokeWeight(1);

  for (let p of particles) {
    // Noise at (x, y) → angle between 0 and 4π
    let angle = noise(p.x * scale, p.y * scale) * TWO_PI * 2;

    let nx = p.x + cos(angle) * 1.5;
    let ny = p.y + sin(angle) * 1.5;

    line(p.x, p.y, nx, ny);

    p.x = nx;
    p.y = ny;

    // New starting position when a particle leaves the canvas
    if (p.x < 0 || p.x > width || p.y < 0 || p.y > height) {
      p.x = random(width);
      p.y = random(height);
    }
  }
}

Noise Landscape

Noise is perfect for terrain and waves:

let t = 0;

function draw() {
  background(30, 100, 180);  // Sky

  fill(50, 180, 80);  // Green terrain
  noStroke();
  beginShape();
  vertex(0, height);

  for (let x = 0; x <= width; x += 5) {
    let y = noise(x * 0.003 + t) * height * 0.7 + height * 0.2;
    vertex(x, y);
  }

  vertex(width, height);
  endShape(CLOSE);

  t += 0.003;  // Let terrain slowly scroll
}

Example: Flow field

Particles follow the angles that noise(x, y) returns at each location. Together, their trails form flowing curves.

Try it yourself

Change the scale (a larger number = finer swirls), tweak the number of particles, or swap the colour. Try noiseSeed() for a fixed field.
