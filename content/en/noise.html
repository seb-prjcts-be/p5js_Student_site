Perlin noise is a special mathematical function that generates smooth, "organic" randomness. Where random() gives a completely random value each frame, noise() gives values that transition smoothly into each other - ideal for fluid movement, textures, and landscapes.

Origin: Ken Perlin

Ken Perlin developed this function in 1983 for the film Tron. He needed a way to generate natural-looking textures for the computer imagery in the movie. His solution - a kind of "smooth" randomness - proved so powerful that he received an Academy Award (Technical Achievement Oscar) for it in 1997.

Since then, Perlin noise has become a standard technique in games, film, and creative coding. In p5.js you use it via the noise() function.

noise() vs random()

The difference is immediately visible:

// random(): jumps to a random value every frame
let x = random(0, width);   // Jerky, restless

// noise(): smooth transition between values
let offset = 0;

function draw() {
  let x = noise(offset) * width;  // noise() returns value between 0 and 1
  offset += 0.01;                 // Small step = smooth motion
}

noise(t) always returns a value between 0 and 1. By slowly increasing t, you move through the noise space.

Basic usage

let t = 0;

function draw() {
  background(245);

  let x = noise(t) * width;
  let y = noise(t + 100) * height;  // Different starting point = different motion

  circle(x, y, 40);

  t += 0.01;  // Speed: smaller = slower and smoother
}

By using t + 100 for the y-coordinate, we ensure that x and y move independently of each other.

Mapping noise() to a Range

noise() always returns a value between 0 and 1. Use map() to convert it to a desired range:

let t = 0;

function draw() {
  let diameter = map(noise(t), 0, 1, 20, 200);
  // or: let diameter = noise(t) * 200;

  background(245);
  circle(width/2, height/2, diameter);
  t += 0.02;
}

2D Noise: Textures

noise() also works in two dimensions: noise(x, y). This way you create seamless textures and "clouds":

function setup() {
  createCanvas(400, 400);
  noLoop();  // Draw only once
}

function draw() {
  loadPixels();
  let scale = 0.005;  // Smaller = larger "blobs"

  for (let x = 0; x < width; x++) {
    for (let y = 0; y < height; y++) {
      let n = noise(x * scale, y * scale);
      let brightness = n * 255;
      let index = (x + y * width) * 4;
      pixels[index]     = brightness;  // R
      pixels[index + 1] = brightness;  // G
      pixels[index + 2] = brightness;  // B
      pixels[index + 3] = 255;         // Alpha
    }
  }
  updatePixels();
}

noiseSeed() and noiseDetail()

You can influence the noise pattern:

// noiseSeed(): same seed = always the same pattern
noiseSeed(42);

// noiseDetail(): how many "layers" of detail (octaves)
// More octaves = more detail but more erratic
noiseDetail(8, 0.5);  // 8 octaves, amplitude halves per octave

Noise for Smooth Movement

A classic: multiple objects each moving independently through noise space:

let offsets = [];
let N = 20;

function setup() {
  createCanvas(600, 400);
  // Give each object a different starting point in noise space
  for (let i = 0; i < N; i++) {
    offsets.push({ x: random(1000), y: random(1000) });
  }
}

function draw() {
  background(30, 20);

  noStroke();
  fill(0, 0, 255, 180);

  for (let off of offsets) {
    let x = noise(off.x) * width;
    let y = noise(off.y) * height;

    circle(x, y, 15);

    off.x += 0.007;  // Move through noise space
    off.y += 0.007;
  }
}

Noise Landscape

Noise is perfect for terrain and waves:

let t = 0;

function draw() {
  background(30, 100, 180);  // Sky

  fill(50, 180, 80);  // Green terrain
  noStroke();
  beginShape();
  vertex(0, height);

  for (let x = 0; x <= width; x += 5) {
    let y = noise(x * 0.003 + t) * height * 0.7 + height * 0.2;
    vertex(x, y);
  }

  vertex(width, height);
  endShape(CLOSE);

  t += 0.003;  // Let terrain slowly scroll
}

Flow Field: noise as a compass

A flow field uses noise to determine a direction at every point on the canvas. Imagine an invisible wind pattern: every particle moving through it follows the local wind direction.

The principle:

  • Divide the canvas into a grid of cells
  • Calculate an angle per cell using noise(x, y)
  • Let particles move in the direction of that angle
let particles = [];

function setup() {
  createCanvas(600, 400);
  for (let i = 0; i < 200; i++) {
    particles.push(createVector(random(width), random(height)));
  }
}

function draw() {
  background(10, 15);
  for (let p of particles) {
    // Noise determines the direction at this position
    let angle = noise(p.x * 0.005, p.y * 0.005, frameCount * 0.003) * TWO_PI * 2;
    p.x += cos(angle);
    p.y += sin(angle);

    // Wrap around edges
    if (p.x > width) p.x = 0;
    if (p.x < 0) p.x = width;
    if (p.y > height) p.y = 0;
    if (p.y < 0) p.y = height;

    stroke(200, 220, 255, 40);
    point(p.x, p.y);
  }
}

Flow fields are the foundation of many generative artworks. By adjusting the noise scale, speed, and number of particles, very different patterns emerge.
