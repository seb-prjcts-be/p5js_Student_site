With p5.js you can draw text on the canvas. You control the size, font, alignment, and color yourself. Text is a powerful component of generative art, data visualization, and interactive sketches.

Basics: Drawing Text

Use text() to draw text. Always set a size with textSize().

function setup() {
  createCanvas(600, 400);
  background(240);
}

function draw() {
  background(240);

  fill(50);          // text color
  noStroke();        // no outline around the text
  textSize(32);      // font size in pixels

  text('Hello world!', 50, 100); // text, x, y
}

The x and y in text() indicate the bottom-left of the text (the baseline), not the top-left corner.

Font Size and Style

function draw() {
  background(240);
  fill(30);
  noStroke();

  // Different sizes
  textSize(12);
  text('Small font', 50, 80);

  textSize(24);
  text('Normal font', 50, 130);

  textSize(48);
  text('Large font', 50, 200);

  // Font style
  textStyle(NORMAL);
  text('Normal', 50, 270);

  textStyle(BOLD);
  text('Bold', 50, 310);

  textStyle(ITALIC);
  text('Italic', 50, 350);

  textStyle(BOLDITALIC);
  text('Bold and italic', 200, 350);
}

Alignment

With textAlign() you control how text is aligned relative to the given position.

function draw() {
  background(240);
  fill(30);
  noStroke();
  textSize(20);

  // Horizontal alignment
  textAlign(LEFT);
  text('Left aligned', 300, 80);

  textAlign(CENTER);
  text('Centered', 300, 130);

  textAlign(RIGHT);
  text('Right aligned', 300, 180);

  // Vertical alignment (second parameter)
  textAlign(CENTER, TOP);
  text('Top', 300, 230);

  textAlign(CENTER, CENTER);
  text('Middle', 300, 280);

  textAlign(CENTER, BOTTOM);
  text('Bottom', 300, 330);

  // Guide line to see the position
  stroke(255, 0, 0);
  line(300, 0, 300, height);
}

Centering text on the canvas

function setup() {
  createCanvas(600, 400);
  textSize(40);
  textAlign(CENTER, CENTER);
  fill(50);
  noStroke();
}

function draw() {
  background(240);
  text('I am in the center!', width / 2, height / 2);
}

Text in a Rectangle

If you pass a fourth and fifth parameter to text(), the text is automatically wrapped within the specified width and height.

function draw() {
  background(240);
  fill(30);
  noStroke();
  textSize(16);

  // text(content, x, y, width, height)
  text('This is a longer text that automatically wraps to the next line when it becomes too long.', 50, 50, 300, 200);

  // Draw frame to see the text box
  noFill();
  stroke(200, 0, 0);
  rect(50, 50, 300, 200);
}

Line Spacing

textSize(18);
textLeading(30); // distance between lines in pixels

text('Line one\nLine two\nLine three', 50, 100);

Measuring Text Width

With textWidth() you can measure how wide a text is. Useful for positioning text precisely.

function draw() {
  background(240);
  textSize(32);
  fill(30);

  let word = 'p5.js';
  let w = textWidth(word);

  // Draw a marker under the word
  text(word, 100, 100);
  stroke(255, 0, 0);
  line(100, 105, 100 + w, 105);

  // Center text manually
  noStroke();
  text(word, width / 2 - w / 2, 200);
}

Loading Custom Fonts

You can load your own font with loadFont(). In modern p5.js you usually do this with await loadFont() in async function setup(). Older tutorials often still use preload() for this.

let myFont;

async function setup() {
  // Load a local font file
  myFont = await loadFont('fonts/my-font.ttf');
  createCanvas(600, 400);
  textFont(myFont); // set the font
}

function draw() {
  background(240);
  textSize(48);
  fill(50);
  text('Custom font!', 50, 200);
}

Tip: Download the font file and save it in your project folder — that's the most reliable approach.

Google Fonts via index.html

The easiest way to use a Google Font is to import it via index.html. This way the browser loads the font automatically when the page opens.

Step 1 — choose a font on Google Fonts

Go to fonts.google.com, find a font and click Get font → Get embed code. You'll get a <link> tag like this:

<link rel="stylesheet" href="https://fonts.googleapis.com/css2?family=Press+Start+2P&display=swap">

Step 2 — paste the link in index.html

Add that line in the <head> of index.html, before style.css:

<head>
  ...
  <link rel="stylesheet" href="https://fonts.googleapis.com/css2?family=Press+Start+2P&display=swap">
  <link rel="stylesheet" href="style.css">
</head>

Step 3 — use the font in your sketch

Call the font by its exact name (as listed on Google Fonts) via textFont():

function setup() {
  createCanvas(600, 400);
  textFont('Press Start 2P');
  textSize(16);
}

function draw() {
  background(245);
  fill(0);
  text('Game Over', 160, 200);
}

Note: the font must be loaded before your sketch uses it. If you import the font in index.html, it's already available when p5.js starts up during normal use. If you're working with a standalone sketch (separate HTML file), add the <link> tag to that file.

Which fonts work well in p5.js?

  • Press Start 2P — pixel font, retro game style
  • Oswald — narrow and strong, easy to read on canvas
  • Space Mono — monospace with character
  • Bebas Neue — wide capitals, strong for titles
  • DM Serif Display — elegant serif for poetic sketches

Text as Generative Art

Text is a powerful medium for generative art. Here are a few ideas:

Moving text

function draw() {
  background(20);
  fill(255);
  textSize(32);
  textAlign(CENTER, CENTER);

  let x = width / 2 + sin(frameCount * 0.05) * 100;
  let y = height / 2 + cos(frameCount * 0.03) * 80;

  text('p5.js', x, y);
}

Text with variable size

function draw() {
  background(240);
  fill(50, 100, 200);
  textAlign(CENTER, CENTER);

  let size = map(mouseX, 0, width, 10, 100);
  textSize(size);
  text('Move the mouse!', width / 2, height / 2);
}

Letters on a circle

let sentence = 'p5.js is awesome! ';

function draw() {
  background(240);
  textSize(18);
  textAlign(CENTER, CENTER);
  fill(30);
  noStroke();

  for (let i = 0; i < sentence.length; i++) {
    let angle = map(i, 0, sentence.length, 0, TWO_PI);
    angle += frameCount * 0.01; // rotate slowly

    let x = width / 2 + cos(angle) * 150;
    let y = height / 2 + sin(angle) * 150;

    push();
    translate(x, y);
    rotate(angle + HALF_PI); // rotate letter in the right direction
    text(sentence[i], 0, 0);
    pop();
  }
}
