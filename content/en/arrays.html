Arrays are lists of values. They are perfect for storing multiple related elements, like positions of balls, colors, or objects in an animation.

Creating Arrays

You create an array like this:

let numbers = [10, 20, 30, 40];
let names = ["red", "green", "blue"];
let empty = [];  // Empty array

// Access via index (starts at 0)
console.log(numbers[0]);  // 10
console.log(names[2]);    // "blue"

Array length

let numbers = [10, 20, 30, 40];
console.log(numbers.length);  // 4

Adding and Removing Elements

Use push() to add elements to the end:

let balls = [];

function mousePressed() {
  balls.push(mouseX);  // Add mouse position
}

console.log(balls);  // [120, 340, 55, ...]

Other useful methods:

  • push(value) — add to the end
  • pop() — remove the last element (and return it)
  • splice(index, count) — remove elements at a position
  • indexOf(value) — find the index of a value

Looping Through Arrays

Use a for loop to process each element:

let positions = [50, 150, 250, 350, 450];

function draw() {
  background(245);
  for (let i = 0; i < positions.length; i++) {
    circle(positions[i], height / 2, 40);
  }
}

Or the modern for...of loop — shorter and more readable:

for (let x of positions) {
  circle(x, height / 2, 40);
}

Object Arrays

Arrays can also contain objects. This lets you store multiple properties per item:

let balls = [
  { x: 100, y: 200, size: 40, color: [255, 0, 0] },
  { x: 300, y: 150, size: 60, color: [0, 0, 255] },
  { x: 450, y: 300, size: 30, color: [0, 255, 0] }
];

function draw() {
  background(30);
  for (let ball of balls) {
    fill(ball.color);
    circle(ball.x, ball.y, ball.size);
  }
}

This is the basis of a particle system: an array of objects that each track their own properties.

Dynamic Object Arrays

You can create and add objects while your sketch is running:

let balls = [];

function mousePressed() {
  // Add a new object on click
  balls.push({
    x: mouseX,
    y: mouseY,
    vx: random(-2, 2),   // Horizontal speed
    vy: random(-4, -1),  // Vertical speed (upward)
    size: random(15, 40)
  });
}

function draw() {
  background(245);

  for (let ball of balls) {
    ball.x += ball.vx;
    ball.y += ball.vy;
    ball.vy += 0.15;  // Gravity

    circle(ball.x, ball.y, ball.size);
  }

  // Remove balls that are off screen
  balls = balls.filter(ball => ball.y < height + 50);
}

Useful Array Methods

filter() — select elements

// Remove dead particles (life <= 0)
particles = particles.filter(p => p.life > 0);

map() — transform each element

let numbers = [1, 2, 3, 4, 5];
let doubled = numbers.map(n => n * 2);
// doubled = [2, 4, 6, 8, 10]

forEach() — do something with each element

balls.forEach(ball => {
  ball.x += ball.vx;
  ball.y += ball.vy;
});
