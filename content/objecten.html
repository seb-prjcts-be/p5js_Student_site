Classes zijn blauwdrukken voor objecten. Ze bundelen eigenschappen (data) en gedrag (functies) in één geheel. In p5.js gebruik je classes vaak voor dingen als ballen, deeltjes of vijanden in een spel — objecten die elk hun eigen toestand bijhouden.

Van Object Literal naar Class

Je kende al object literals vanuit arrays:

let bal = { x: 100, y: 200, vx: 2, vy: 0 };

Maar wat als je tientallen ballen wilt, elk met eigen methoden? Dan is een class handiger:

class Bal {
  constructor(x, y) {
    this.x = x;
    this.y = y;
    this.vx = random(-2, 2);
    this.vy = random(-3, -1);
  }

  update() {
    this.x += this.vx;
    this.y += this.vy;
    this.vy += 0.1;  // Zwaartekracht
  }

  teken() {
    circle(this.x, this.y, 20);
  }
}

Objecten Aanmaken met new

Je maakt een instantie van een class met new:

let mijnBal = new Bal(200, 300);  // Roept de constructor aan

mijnBal.update();  // Voer een methode uit
mijnBal.teken();

Meerdere Objecten in een Array

let ballen = [];

function setup() {
  createCanvas(600, 400);

  // Maak 10 ballen aan
  for (let i = 0; i < 10; i++) {
    ballen.push(new Bal(random(width), random(height)));
  }
}

function draw() {
  background(220);

  for (let bal of ballen) {
    bal.update();
    bal.teken();
  }
}

Een Volledig Class Voorbeeld

Een bal die stuitert tegen de randen van het canvas:

class StuitendeBal {
  constructor(x, y, diameter) {
    this.x = x;
    this.y = y;
    this.d = diameter;
    this.vx = random(-3, 3);
    this.vy = random(-3, 3);
    this.kleur = color(random(255), random(255), random(255));
  }

  update() {
    this.x += this.vx;
    this.y += this.vy;

    // Stuit tegen linker/rechter rand
    if (this.x < this.d / 2 || this.x > width - this.d / 2) {
      this.vx *= -1;
    }
    // Stuit tegen boven/onder rand
    if (this.y < this.d / 2 || this.y > height - this.d / 2) {
      this.vy *= -1;
    }
  }

  teken() {
    fill(this.kleur);
    noStroke();
    circle(this.x, this.y, this.d);
  }
}

let ballen = [];

function setup() {
  createCanvas(600, 400);
  for (let i = 0; i < 15; i++) {
    ballen.push(new StuitendeBal(
      random(width), random(height), random(20, 60)
    ));
  }
}

function draw() {
  background(30);
  for (let bal of ballen) {
    bal.update();
    bal.teken();
  }
}

Methoden die Iets Teruggeven

Methoden kunnen ook waarden berekenen en teruggeven:

class Bal {
  constructor(x, y) {
    this.x = x;
    this.y = y;
    this.d = 30;
  }

  // Geeft true als twee ballen elkaar overlappen
  raakvlak(andereBal) {
    let afstand = dist(this.x, this.y, andereBal.x, andereBal.y);
    return afstand < (this.d + andereBal.d) / 2;
  }

  teken(geraakt) {
    fill(geraakt ? 255 : 100, 150, 255);
    circle(this.x, this.y, this.d);
  }
}

let a, b;

function setup() {
  createCanvas(400, 300);
  a = new Bal(100, 150);
  b = new Bal(300, 150);
}

function draw() {
  background(220);
  b.x = mouseX;
  b.y = mouseY;

  let overlapt = a.raakvlak(b);
  a.teken(overlapt);
  b.teken(overlapt);
}

this Sleutelwoord

this verwijst altijd naar het huidige object. Zo onderscheid je eigenschappen van het object van lokale variabelen:

class Teller {
  constructor(startWaarde) {
    this.waarde = startWaarde;  // Eigenschap van het object
  }

  verhoog(stap) {
    let stap = stap || 1;    // Lokale variabele
    this.waarde += stap;     // Werkt op het object
  }
}
