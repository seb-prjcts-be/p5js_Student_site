Met p5.js kun je tekst tekenen op het canvas. Je bepaalt zelf de grootte, het lettertype, de uitlijning en de kleur. Tekst is een krachtig onderdeel van generatieve kunst, data visualisatie en interactieve sketches.

Basis: Tekst Tekenen

Gebruik text() om tekst te tekenen. Stel altijd een grootte in met textSize().

function setup() {
  createCanvas(600, 400);
  background(240);
}

function draw() {
  background(240);

  fill(50);          // kleur van de tekst
  noStroke();        // geen rand om de tekst
  textSize(32);      // lettergrootte in pixels

  text('Hallo wereld!', 50, 100); // tekst, x, y
}

De x en y in text() geven de onderkant links van de tekst aan (de baseline), niet de linkerbovenhoek.

Lettergrootte en Stijl

function draw() {
  background(240);
  fill(30);
  noStroke();

  // Verschillende groottes
  textSize(12);
  text('Klein lettertype', 50, 80);

  textSize(24);
  text('Normaal lettertype', 50, 130);

  textSize(48);
  text('Groot lettertype', 50, 200);

  // Lettertype stijl
  textStyle(NORMAL);
  text('Normaal', 50, 270);

  textStyle(BOLD);
  text('Vet', 50, 310);

  textStyle(ITALIC);
  text('Cursief', 50, 350);

  textStyle(BOLDITALIC);
  text('Vet en cursief', 200, 350);
}

Uitlijning

Met textAlign() bepaal je hoe tekst wordt uitgelijnd ten opzichte van de opgegeven positie.

function draw() {
  background(240);
  fill(30);
  noStroke();
  textSize(20);

  // Horizontale uitlijning
  textAlign(LEFT);
  text('Links uitgelijnd', 300, 80);

  textAlign(CENTER);
  text('Gecentreerd', 300, 130);

  textAlign(RIGHT);
  text('Rechts uitgelijnd', 300, 180);

  // Verticale uitlijning (tweede parameter)
  textAlign(CENTER, TOP);
  text('Boven', 300, 230);

  textAlign(CENTER, CENTER);
  text('Midden', 300, 280);

  textAlign(CENTER, BOTTOM);
  text('Onder', 300, 330);

  // Hulplijn om de positie te zien
  stroke(255, 0, 0);
  line(300, 0, 300, height);
}

Tekst centreren op het canvas

function setup() {
  createCanvas(600, 400);
  textSize(40);
  textAlign(CENTER, CENTER);
  fill(50);
  noStroke();
}

function draw() {
  background(240);
  text('Ik sta in het midden!', width / 2, height / 2);
}

Tekst in een Rechthoek

Als je een vierde en vijfde parameter meegeeft aan text(), wordt de tekst automatisch afgebroken binnen de opgegeven breedte en hoogte.

function draw() {
  background(240);
  fill(30);
  noStroke();
  textSize(16);

  // text(inhoud, x, y, breedte, hoogte)
  text('Dit is een langere tekst die automatisch op de volgende regel doorloopt als hij te lang wordt.', 50, 50, 300, 200);

  // Kader tekenen om de tekstbox te zien
  noFill();
  stroke(200, 0, 0);
  rect(50, 50, 300, 200);
}

Regelafstand

textSize(18);
textLeading(30); // afstand tussen regels in pixels

text('Regel één\nRegel twee\nRegel drie', 50, 100);

Tekstbreedte Meten

Met textWidth() kun je meten hoe breed een tekst is. Handig om tekst precies te positioneren.

function draw() {
  background(240);
  textSize(32);
  fill(30);

  let woord = 'p5.js';
  let breedte = textWidth(woord);

  // Teken een markering onder het woord
  text(woord, 100, 100);
  stroke(255, 0, 0);
  line(100, 105, 100 + breedte, 105);

  // Tekst precies centreren (handmatig)
  noStroke();
  text(woord, width / 2 - breedte / 2, 200);
}

Custom Fonts Laden

Je kunt je eigen lettertype laden met loadFont(). In moderne p5.js doe je dat meestal met await loadFont() in async function setup(). Oudere tutorials gebruiken hier vaak nog preload().

let mijnFont;

async function setup() {
  // Laad een lokaal font bestand
  mijnFont = await loadFont('fonts/mijn-lettertype.ttf');
  createCanvas(600, 400);
  textFont(mijnFont); // stel het font in
}

function draw() {
  background(240);
  textSize(48);
  fill(50);
  text('Custom lettertype!', 50, 200);
}

Tip: Download het font bestand en sla het op in je projectmap - dat is het meest betrouwbaar.

Google Fonts via index.html

De makkelijkste manier om een Google Font te gebruiken is het importeren via index.html. Zo laadt de browser het font automatisch bij het openen van de pagina.

Stap 1 - kies een font op Google Fonts

Ga naar fonts.google.com, zoek een font en klik op Get font → Get embed code. Je krijgt een <link>-tag zoals deze:

<link rel="stylesheet" href="https://fonts.googleapis.com/css2?family=Press+Start+2P&display=swap">

Stap 2 - plak de link in index.html

Voeg die regel toe in de <head> van index.html, vóór style.css:

<head>
  ...
  <link rel="stylesheet" href="https://fonts.googleapis.com/css2?family=Press+Start+2P&display=swap">
  <link rel="stylesheet" href="style.css">
</head>

Stap 3 - gebruik het font in je sketch

Roep het font aan met zijn exacte naam (zoals op Google Fonts vermeld) via textFont():

function setup() {
  createCanvas(600, 400);
  textFont('Press Start 2P');
  textSize(16);
}

function draw() {
  background(245);
  fill(0);
  text('Game Over', 160, 200);
}

Let op: het font moet geladen zijn vóór je sketch het gebruikt. Als je het font in index.html importeert, is het bij normaal gebruik al beschikbaar wanneer p5.js opstart. Als je werkt met een lokale schets (los HTML-bestand), voeg je de <link>-tag toe aan dat bestand.

Welke fonts werken goed in p5.js?

  • Press Start 2P - pixel-lettertype, retro game-stijl
  • Oswald - smal en krachtig, goed leesbaar op canvas
  • Space Mono - monospace met karakter
  • Bebas Neue - brede kapitalen, sterk voor titels
  • DM Serif Display - elegant serif voor poëtische sketches

Tekst als Generatieve Kunst

Tekst is een krachtig medium voor generatieve kunst. Hier zijn een paar ideeën:

Bewegende tekst

function draw() {
  background(20);
  fill(255);
  textSize(32);
  textAlign(CENTER, CENTER);

  let x = width / 2 + sin(frameCount * 0.05) * 100;
  let y = height / 2 + cos(frameCount * 0.03) * 80;

  text('p5.js', x, y);
}

Tekst met variabele grootte

function draw() {
  background(240);
  fill(50, 100, 200);
  textAlign(CENTER, CENTER);

  let grootte = map(mouseX, 0, width, 10, 100);
  textSize(grootte);
  text('Beweeg de muis!', width / 2, height / 2);
}

Letters op een cirkel

let zin = 'p5.js is geweldig! ';

function draw() {
  background(240);
  textSize(18);
  textAlign(CENTER, CENTER);
  fill(30);
  noStroke();

  for (let i = 0; i < zin.length; i++) {
    let hoek = map(i, 0, zin.length, 0, TWO_PI);
    hoek += frameCount * 0.01; // draai langzaam

    let x = width / 2 + cos(hoek) * 150;
    let y = height / 2 + sin(hoek) * 150;

    push();
    translate(x, y);
    rotate(hoek + HALF_PI); // letter in de juiste richting draaien
    text(zin[i], 0, 0);
    pop();
  }
}
