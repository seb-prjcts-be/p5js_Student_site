Functies zijn herbruikbare stukken code. Ze maken je code overzichtelijker, korter en makkelijker te onderhouden. In p5.js gebruik je ze voortdurend: setup() en draw() zijn zelf ook functies!

Functies Definiëren en Aanroepen

Een functie definieer je met het sleutelwoord function:

// Definitie
function tekenCirkel() {
  ellipse(200, 200, 50, 50);
}

// Aanroepen — doe dit in draw() of setup()
function draw() {
  background(220);
  tekenCirkel();  // Voer de functie uit
  tekenCirkel();  // Nog een keer — code hoef je niet te herhalen
}

Parameters: Waarden Doorgeven

Parameters maken functies flexibel. Je definieert ze tussen de haakjes:

function tekenCirkel(x, y, diameter) {
  ellipse(x, y, diameter, diameter);
}

function draw() {
  background(220);
  tekenCirkel(100, 150, 50);   // x=100, y=150, diameter=50
  tekenCirkel(300, 200, 80);   // Zelfde functie, andere waarden
  tekenCirkel(450, 100, 30);
}

Meerdere Parameters met Kleur

function tekenBal(x, y, diameter, r, g, b) {
  fill(r, g, b);
  noStroke();
  ellipse(x, y, diameter, diameter);
}

function draw() {
  background(30);
  tekenBal(150, 200, 60, 255, 100,  50);  // rood
  tekenBal(300, 200, 80, 50,  200, 255);  // blauw
  tekenBal(450, 200, 40, 100, 255, 100);  // groen
}

Return Waarden

Een functie kan een berekende waarde teruggeven met return:

function berekenAfstand(x1, y1, x2, y2) {
  let dx = x2 - x1;
  let dy = y2 - y1;
  return sqrt(dx * dx + dy * dy);  // Pythagoras
}

function draw() {
  let afstand = berekenAfstand(mouseX, mouseY, width/2, height/2);
  let diameter = map(afstand, 0, 300, 10, 150);

  ellipse(width/2, height/2, diameter, diameter);
}

Nog een voorbeeld: willekeurige kleur

function willekeurigeKleur() {
  return color(random(255), random(255), random(255));
}

function mousePressed() {
  fill(willekeurigeKleur());
  ellipse(mouseX, mouseY, 40, 40);
}

Scope: Waar Leven Variabelen?

Variabelen gedeclareerd in een functie bestaan alleen daarbinnen. Variabelen buiten alle functies zijn globaal beschikbaar:

let globaleX = 0;  // Beschikbaar overal

function draw() {
  let lokaalY = 100;  // Alleen beschikbaar in draw()
  globaleX += 1;

  // lokaalY bestaat niet meer na deze accolade
}

// globaleX is hier nog steeds beschikbaar

Complex Voorbeeld: Bloem Tekenen

Door functies samen te gebruiken kun je complexe vormen eenvoudig hergebruiken:

function tekenBloemblaadje(x, y, hoek, grootte) {
  push();
    translate(x, y);
    rotate(hoek);
    noStroke();
    fill(255, 180, 200, 180);
    ellipse(grootte * 0.6, 0, grootte * 1.2, grootte * 0.6);
  pop();
}

function tekenBloem(x, y, grootte) {
  let aantalBlaadjes = 8;
  for (let i = 0; i < aantalBlaadjes; i++) {
    let hoek = (TWO_PI / aantalBlaadjes) * i;
    tekenBloemblaadje(x, y, hoek, grootte);
  }
  // Midden
  fill(255, 220, 50);
  noStroke();
  ellipse(x, y, grootte * 0.5, grootte * 0.5);
}

function draw() {
  background(200, 230, 200);
  tekenBloem(150, 200, 60);
  tekenBloem(350, 150, 80);
  tekenBloem(500, 250, 50);
}
